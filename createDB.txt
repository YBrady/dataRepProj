create table people (
id int NOT NULL AUTO_INCREMENT,
name varchar(250),
gender varchar(10),
age int,
nice boolean,
toy varchar(250),
chimney boolean,
toyok boolean,
PRIMARY KEY(id));


create table toys(
itemId int NOT NULL AUTO_INCREMENT,
item varchar(250),
stock int,
PRIMARY KEY(itemId));
